Get Started in Minutes

# Install the SDK
$ pip install ohm-lab-sdk

# Initialise your board
$ ohm init --board neuro-n6

# Deploy a model
$ ohm deploy model/lite --target n6

# Run real-time inference
$ ohm run --camera mipi --model/face_estimation
